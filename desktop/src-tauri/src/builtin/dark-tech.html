query.pql · 找出最慢的 5 个接口 
// 过去 1 小时，按 P95 延迟排序
from spans
where service = "api-gateway"
  and timestamp > now() - 1h
  and status_code >= 400
group by route
select p95(duration), count(), rate(error)
order by p95 desc
limit 5
